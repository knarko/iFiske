PSUEDO SQL:

table regions
	int primary key id, text name, real long, real lat, int quantity

table areas
	int primary key id, text name, int foreign key region_id, int foreign key org_id, real long, real lat

table area_keywords
	int foreign key area_id, text keyword

table products
	int primary key id, int smsdisplay, int vat, int saleschannel, int foreign key area_id, text name, int price, int foreign rule_id, int sortorder, text headline, text important, text notes, text smscode

table species_areas
	int foreign key species_id, int foreign key area_id, text cmt, int level

table species
	int primary key id, text name, text latin

table organisations
	int primary key id, text name, text region, text description, text homepage, text contact


SQL SCHEMA:

'DROP TABLE IF EXISTS Regions'
'CREATE TABLE IF NOT EXISTS Regions (id int, name text, long real, lat real, quantity int, PRIMARY KEY (id))'
'DROP TABLE IF EXISTS Areas'
'CREATE TABLE IF NOT EXISTS Areas (id int,name text, region_id int, org_id int,long real,lat real, PRIMARY KEY (id), FOREIGN KEY (region_id) REFERENCES Regions(id), FOREIGN KEY (org_id) REFERENCES Organisations(id))'
'DROP TABLE IF EXISTS Area_keywords'
'CREATE TABLE IF NOT EXISTS Area_keywords (area_id int, keyword text, FOREIGN KEY (area_id) REFERENCES Areas(id))'
'DROP TABLE IF EXISTS Products'
'CREATE TABLE IF NOT EXISTS Products (id int, smsdisplay int, vat int, saleschannel int, area_id int, name text, price int, rule_id int, sortorder int, headline text, important text, notes text, smscode text, PRIMARY KEY (id), FOREIGN KEY (area_id) REFERENCES Areas(id), FOREIGN KEY (rule_id) REFERENCES Rules(id))'
'DROP TABLE IF EXISTS Species_areas'
'CREATE TABLE IF NOT EXISTS Species_areas (species_id int, area_id int, cmt text, level int, FOREIGN KEY (species_id) REFERENCES Species(id), FOREIGN KEY (area_id) REFERENCES Areas(id))'
'DROP TABLE IF EXISTS Species'
'CREATE TABLE IF NOT EXISTS Species (id int, name text, latin text, PRIMARY KEY (id))'
'DROP TABLE IF EXISTS Organisations'
'CREATE TABLE IF NOT EXISTS Organisations (id int, name text, region region, description text, homepage text, contact text, PRIMARY KEY (id))'
